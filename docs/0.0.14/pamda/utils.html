123    def read_csv(self, filename, has_header=True, return_dict=True):
124        """
125        Function:
126
127        - Reads the contents of a csv and converts it to list of dicts or list of lists
128
129        Requires:
130
131        - `filename`:
132            - Type: str
133            - What: The filepath of the csv to read
134
135        Optional:
136
137        - `has_header`:
138            - Type: bool
139            - What: Flag to indicate if the csv has an initial row that identifies columns
140            - Default: True
141            - Note: Returns a list of lists
142        - `return_dict`:
143            - Type: bool
144            - What: Flag to indicate if the csv should be converted to:
145                - True: list of dicts
146                - False: list of lists
147            - Note: If True, requires `has_header` to be True as the header determines the keys of the dicts
148        """
149        with open(filename) as f:
150            file_data = csv.reader(f, delimiter=",", quotechar='"')
151            if has_header:
152                headers = next(file_data)
153            if has_header and return_dict:
154                return [dict(zip(headers, i)) for i in file_data]
155            elif not has_header and return_dict:
156                self.exception(message="If `return_dict` is True, `has_header` must also be True.")
157            else:
158                return [i for i in file_data]

160    def write_csv(self, filename, data):
161        """
162        Function:
163
164        - Writes the contents of a list of list or list of dicts to a csv
165
166        Requires:
167
168        - `filename`:
169            - Type: str
170            - What: The filepath of the csv to read
171        - `data`:
172            - Type: list of lists | list of dicts
173            - What: The data to write
174        """
175        with open(filename, 'w') as f:
176            if isinstance(data[0], dict):
177                writer=csv.DictWriter(f, fieldnames=data[0].keys())
178                writer.writeheader()
179                for row in data:
180                    writer.writerow(row)
181            elif isinstance(data[0], list):
182                writer=csv.writer(f)
183                for row in data:
184                    writer.writerow(row)
185            else:
186                self.exception("write_csv takes in list of lists or list of dicts only.")

259    def getForceDict(self, object, key):
260        """
261        Function:
262
263        - Returns a value from a dictionary given a key and forces that value to be a dictionary
264        - Note: This updates the object in place to force the value from the key to be a dictionary
265
266        Requires:
267
268        - `object`:
269            - Type: dict
270            - What: The object from which to look for a key
271        - `key`:
272            - Type: str
273            - What: The key to look up in the object
274
275        Example:
276
277        ```
278        data = {'a':{}, 'b':1}
279
280        p.getForceDict(data, 'a') #=> {}
281        p.getForceDict(data, 'b') #=> {}
282
283        # Note that the object has been updated in place
284        data #=> {'a':{}, 'b':{}}
285        ```
286        """
287        if not isinstance(object.get(key), dict):
288            object.__setitem__(key, {})
289        return object.get(key)
