  11        """
  12        Function:
  13
  14        - Returns an accumulated list of items by iterating a function starting with an accumulator over a list
  15
  16        Requires:
  17
  18        - `fn`:
  19            - Type: function | method
  20            - What: The function or method to reduce
  21            - Note: This function should have an arity of 2 (take two inputs)
  22            - Note: The first input should take the accumulator value
  23            - Note: The second input should take the data value
  24        -`initial_accumulator`:
  25            - Type: any
  26            - What: The initial item to pass into the function when starting the accumulation process
  27        - `data`:
  28            - Type: list
  29            - What: The list of items to iterate over
  30
  31        Example:
  32
  33        ```
  34        data=[1,2,3,4]
  35        pamda.accumulate(
  36            fn=pamda.add,
  37            initial_accumulator=0,
  38            data=data
  39        )
  40        #=> [1,3,6,10]
  41
  42        ```
  43        """
  44        fn = self.curry(fn)
  45        if fn.__arity__ != 2:
  46            raise Exception("`fn` must have an arity of 2 (take two inputs)")

  48            raise Exception(

  50            )
  51        acc = initial_accumulator
  52        out = []
  53        for i in data:
  54            acc = fn(acc, i)
  55            out.append(acc)

 341    def curry(self, fn):
 342        """
 343        Function:
 344
 345        - Curries a function such that inputs can be added interatively
 346
 347        Requires:
 348
 349        - `fn`:
 350            - Type: function | method
 351            - What: The function or method to curry
 352            - Note: Class methods auto apply self during curry
 353
 354        Notes:
 355
 356        - Once curried, the function | method becomes a curry_obj object
 357        - The initial function is only called once all inputs are passed
 358
 359
 360        Examples:
 361
 362        ```
 363        curriedZip=pamda.curry(pamda.zip)
 364        curriedZip(['a','b'])([1,2]) #=> [['a',1],['b',2]]

 366        # Curried functions can be thunkified at any time
 367        # See also thunkify
 368        zipThunk=curriedZip.thunkify()(['a','b'])([1,2])
 369        zipThunk() #=> [['a',1],['b',2]]
 370        ```
 371
 372        ```
 373        def myFunction(a,b,c):
 374            return [a,b,c]
 375
 376        curriedMyFn=pamda.curry(myFunction)
 377
 378        curriedMyFn(1,2,3) #=> [1,2,3]
 379        curriedMyFn(1)(2,3) #=> [1,2,3]
 380
 381        x=curriedMyFn(1)(2)
 382        x(3) #=> [1,2,3]
 383        x(4) #=> [1,2,4]
 384
 385
 386        ```
 387        """
 388        if fn.__dict__.get("__isCurried__"):
 389            return fn()
 390        return curry_obj(fn)

 534        """
 535        Function:
 536
 537        - Returns a new function equivalent to the supplied function except that the first two inputs are flipped
 538
 539        Requires:
 540
 541        - `fn`:
 542            - Type: function | method
 543            - What: The function or method to flip

 628        """
 629        Function:
 630
 631        - Splits a list into a dictionary of sublists keyed by the return string of a provided function
 632
 633        Requires:
 634
 635        - `fn`:
 636            - Type: function | method
 637            - What: The function or method to group by
 638            - Note: Must return a string
 639            - Note: This function must be unary (take one input)
 640            - Note: This function is applied to each item in the list recursively
 641        - `data`:
 642            - Type: list
 643            - What: List of items to apply the function to and then group by the results
 644
 645        Examples:
 646
 647        ```
 648        def getGrade(item):
 649            score=item['score']
 650            if score>90:
 651                return 'A'
 652            elif score>80:
 653                return 'B'
 654            elif score>70:
 655                return 'C'
 656            elif score>60:
 657                return 'D'
 658            else:
 659                return 'F'
 660

 662            {'name':'Connor', 'score':75},
 663            {'name':'Fred', 'score':79},
 664            {'name':'Joe', 'score':84},

 669        #=>    'C':[{'name':'Connor', 'score':75},{'name':'Fred', 'score':79}]
 670        #=>}
 671        ```
 672        """
 673        curried_fn = self.curry(fn)
 674        if curried_fn.__arity__ != 1:
 675            raise Exception(
 676                "groupBy `fn` must only take one parameter as its input"
 677            )

 680            path = fn(i)
 681            if not isinstance(path, str):
 682                raise Exception(
 683                    f"groupBy `fn` must return a str but instead returned {path}"
 684                )
 685            if path not in output:
 686                output[path] = []
 687            output[path].append(i)
 688        return output

1025    def mergeDeep(self, update_data, data):
1026        """
1027        Function:
1028
1029        - Recursively merges two nested dictionaries keeping all keys at each layer
1030        - Values from `update_data` are used when keys are present in both dictionaries
1031
1032        Requires:
1033
1034        - `update_data`:
1035            - Type: any
1036            - What: The new data that will take precedence during merging
1037        - `data`:
1038            - Type: any
